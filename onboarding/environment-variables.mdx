For reference, this static analysis was done and generated by the following env-validator script:


```bash
#!/usr/bin/env bash
# ───────────────────────────────────────────────────────────────
# PW-3757: Client Environment Variable Usage Analysis
#
# Scans client/.env + client/.env.example for all defined vars,
# then searches client/ for every USAGE of each one - excluding
# definition sites (schemas, type declarations, .env files).
#
# Usage: bash scripts/analyze-client-env-usage.sh
# Output: Markdown to stdout (pipe to .md file if desired)
# ───────────────────────────────────────────────────────────────

set -euo pipefail

REPO_ROOT="$(cd "$(dirname "$0")/.." && pwd)"
CLIENT_DIR="$REPO_ROOT/client"

# ── Collect all env var names ──

collect_vars_from_file() {
 local file="$1"
 if [[ -f "$file" ]]; then
 grep -oP '^[A-Z][A-Z0-9_]+(?==)' "$file" 2>/dev/null || true
 fi
}

collect_vars_from_code() {
 grep -roPh 'process\.env\.([A-Z][A-Z0-9_]+)' "$CLIENT_DIR/src" "$CLIENT_DIR/next.config.ts" \
 --include='*.ts' --include='*.tsx' --include='*.js' --include='*.mjs' \
 2>/dev/null | sed 's/process\.env\.//' | sort -u || true
}

echo "Scanning for environment variables..." >&2

ALL_VARS=$(
 {
 collect_vars_from_file "$CLIENT_DIR/.env"
 collect_vars_from_file "$CLIENT_DIR/.env.example"
 collect_vars_from_code
 } | sort -u
)

TOTAL=$(echo "$ALL_VARS" | wc -l)
echo "Found $TOTAL unique env vars. Searching for usages..." >&2

# ── Auto-ingested vars (read by SDKs without explicit process.env in our code) ──
# These are hardcoded because static analysis can't detect SDK internals.

AUTO_INGESTED_VARS=(
 "DATABASE_URL:Prisma:Auto-read for database connection string"
 "DIRECT_URL:Prisma:Auto-read for direct (non-pooled) database connection"
 "AUTH_TRUST_HOST:Better Auth:Auto-read to trust the host header"
 "ANTHROPIC_API_KEY:@ai-sdk/anthropic:Auto-read by Anthropic AI SDK"
 "LIVEKIT_API_KEY:livekit-server-sdk:Auto-read by LiveKit server SDK"
 "LIVEKIT_API_SECRET:livekit-server-sdk:Auto-read by LiveKit server SDK"
 "STRIPE_SECRET_KEY:stripe:Auto-read by Stripe Node SDK"
 "STRIPE_WEBHOOK_SECRET:stripe:Used by Stripe webhook signature verification"
 "PORT:Next.js:Auto-read for dev server port"
 "DAILY_API_KEY:@daily-co/daily-js:May be auto-read by Daily.co SDK"
)

# Build a lookup set of auto-ingested var names
declare -A AUTO_INGESTED_SET
for entry in "${AUTO_INGESTED_VARS[@]}"; do
 varname="${entry%%:*}"
 AUTO_INGESTED_SET["$varname"]=1
done

# ── Analyze each var ──

USED_ENTRIES=""
UNUSED_ENTRIES=""
USED_COUNT=0
UNUSED_COUNT=0
IDX=0

while IFS= read -r VAR; do
 [[ -z "$VAR" ]] && continue
 IDX=$((IDX + 1))

 # Search all of client/ with word boundaries, then EXCLUDE definition-only sites:
 # - node_modules, .next (build artifacts)
 # - .env, .env.example, .env.* (where vars are defined)
 # - schemas/ dir (Zod schema declarations)
 # - environment.d.ts (TS type declarations)
 # - env.schema.ts (merged schema)
 # - validated-env.ts (validation bootstrap)
 HITS=$(
 grep -rn -P "\b${VAR}\b" "$CLIENT_DIR" \
 --include='*.ts' --include='*.tsx' --include='*.js' --include='*.mjs' \
 2>/dev/null \
 | grep -v 'node_modules' \
 | grep -v '\.next/' \
 | grep -v '/schemas/.*-env\.schema\.ts' \
 | grep -v 'env\.schema\.ts' \
 | grep -v 'validated-env\.ts' \
 | grep -v 'environment\.d\.ts' \
 || true
 )

 if [[ -z "$HITS" ]]; then
 # Skip auto-ingested vars - they get their own section
 if [[ -z "${AUTO_INGESTED_SET[$VAR]+x}" ]]; then
 UNUSED_COUNT=$((UNUSED_COUNT + 1))
 UNUSED_ENTRIES+="| $UNUSED_COUNT | \`$VAR\` |"$'\n'
 fi
 else
 USED_COUNT=$((USED_COUNT + 1))

 # Get unique files with line numbers
 FILE_LIST=$(
 echo "$HITS" | while IFS= read -r line; do
 echo "$line" | sed "s|^$REPO_ROOT/||" | cut -d: -f1-2
 done | sort -u
 )

 # Format as backtick-wrapped, comma-separated locations
 FORMATTED=$(echo "$FILE_LIST" | sed 's/.*$/`&`/' | paste -sd ',' - | sed 's/,/, /g')

 USED_ENTRIES+="| $USED_COUNT | \`$VAR\` | $FORMATTED |"$'\n'
 fi

 if (( IDX % 20 == 0 )); then
 echo " Processed $IDX / $TOTAL vars..." >&2
 fi
done <<< "$ALL_VARS"

# ── Count auto-ingested vars that are actually defined in .env/.env.example ──

AUTO_INGESTED_ENTRIES=""
AUTO_INGESTED_COUNT=0
for entry in "${AUTO_INGESTED_VARS[@]}"; do
 varname="${entry%%:*}"
 rest="${entry#*:}"
 sdk="${rest%%:*}"
 note="${rest#*:}"
 if echo "$ALL_VARS" | grep -qx "$varname"; then
 AUTO_INGESTED_COUNT=$((AUTO_INGESTED_COUNT + 1))
 AUTO_INGESTED_ENTRIES+="| $AUTO_INGESTED_COUNT | \`$varname\` | $sdk | $note |"$'\n'
 fi
done

# ── Output Report ──

echo ""
echo "# Client Environment Variable Usage Report"
echo ""
echo "_Generated: $(date -u +%Y-%m-%dT%H:%M:%SZ) | Branch: $(git rev-parse --abbrev-ref HEAD)_"
echo ""
echo "## Summary"
echo ""
echo "| Metric | Count |"
echo "|--------|-------|"
echo "| Total vars defined | $TOTAL |"
echo "| **Used in client/** | **$USED_COUNT** |"
echo "| **MAYBE used (auto-ingested by SDKs)** | **$AUTO_INGESTED_COUNT** |"
echo "| **NOT used in client/** | **$UNUSED_COUNT** |"
echo ""

echo "---"
echo ""
echo "## Variables USED in client/ ($USED_COUNT)"
echo ""
echo "| # | Variable | Locations (file:line) |"
echo "|---|----------|-----------------------|"
echo -n "$USED_ENTRIES"

echo ""
echo "---"
echo ""
echo "## Variables MAYBE used in client/ - Auto-Ingested ($AUTO_INGESTED_COUNT)"
echo ""
echo "These vars have **no explicit \`process.env\` reference** in client code, but are"
echo "**auto-read by installed SDKs** at runtime. Static analysis cannot detect this usage."
echo ""
echo "| # | Variable | SDK / Package | Notes |"
echo "|---|----------|---------------|-------|"
echo -n "$AUTO_INGESTED_ENTRIES"

echo ""
echo "---"
echo ""
echo "## Variables NOT USED in client/ ($UNUSED_COUNT)"
echo ""
echo "Defined in \`.env\` / \`.env.example\` but **zero usage** in \`client/\` application code."
echo "_(Excluded from search: Zod schemas, environment.d.ts, validated-env.ts, .env files)_"
echo ""
echo "| # | Variable |"
echo "|---|----------|"
echo -n "$UNUSED_ENTRIES"

echo ""
echo "---"
echo ""
echo "_Search scope: all \`client/**/*.{ts,tsx,js,mjs}\` excluding node_modules, .next, schema definitions, and type declarations_"
```
